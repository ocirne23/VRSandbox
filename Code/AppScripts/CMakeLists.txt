project(AppScripts)

# The "App-Scripts" module (see the SCRIPTS_STATIC option in the root CMakeLists). In a cooked build it generates
# ONE aggregate TU that #includes every Assets/Scripts/*.scr AND *.dsl into its own namespace and compiles it
# into the engine, linking every engine library like App — so the scripts' ctx->* calls resolve to the real
# thunks and get whole-program-optimized (/GL + /LTCG) and inlined. A .dsl is just as includable as a .scr: it
# holds real body-only C++ up top (the DSL editor writes it there on every save, see Transpiler.ixx), the
# "//@@dsl ... //@@end" block below it is ordinary comments as far as the compiler's concerned. Either kind ends
# with REGISTER_*() markers (see ScriptAPI.h) that register the functions it defines; the aggregate needs no
# parsing. App links this target. Off (default), it is an empty stub so App's link line names it unconditionally
# and the runtime-DLL script path (ScriptHost) stays the only one in play.
#
# Adding/renaming/removing a script needs a reconfigure (the glob + generated aggregate change). Editing a
# script's body just recompiles the aggregate (each file is an #include dependency of the generated TU).

if(SCRIPTS_STATIC)
	file(GLOB SCRIPT_FILES "${CMAKE_SOURCE_DIR}/Assets/Scripts/*.scr" "${CMAKE_SOURCE_DIR}/Assets/Scripts/*.dsl")

	set(AGG "// GENERATED by Code/AppScripts/CMakeLists.txt — do not edit.\n")
	string(APPEND AGG "#define SCRIPT_STATIC_BUILD 1\n")
	string(APPEND AGG "#include \"ScriptAPI.h\"\n\n") # ABI header (layout-mirror Entity + glm), same as the DLL path — no module import, so glm isn't pulled in twice
	string(APPEND AGG "#pragma warning(push)\n")
	string(APPEND AGG "#pragma warning(disable: 4505) // unreferenced function with internal linkage has been removed \n\n")
	foreach(scr ${SCRIPT_FILES})
		get_filename_component(stem ${scr} NAME_WE)
		get_filename_component(ext ${scr} EXT) # ".scr" or ".dsl" -- VR_CURRENT_SCRIPT must match the REAL asset
		                                        # path (ScriptComponent::scriptPath/findStaticEntries' lookup), not
		                                        # a hardcoded extension
		string(APPEND AGG "#define VR_CURRENT_SCRIPT \"Scripts/${stem}${ext}\"\n")
		string(APPEND AGG "namespace Script_${stem} {\n")
		string(APPEND AGG "#include \"${scr}\"\n")
		string(APPEND AGG "}\n")
		string(APPEND AGG "#undef VR_CURRENT_SCRIPT\n\n")
	endforeach()
	string(APPEND AGG "#pragma warning(pop)\n")

	set(AGG_FILE "${CMAKE_CURRENT_BINARY_DIR}/StaticScripts.gen.cpp")
	file(GENERATE OUTPUT ${AGG_FILE} CONTENT "${AGG}")

	add_library(${PROJECT_NAME} STATIC ${AGG_FILE})
	set_source_files_properties(${AGG_FILE} PROPERTIES OBJECT_DEPENDS "${SCRIPT_FILES}") # recompile when a script's body changes

	target_include_directories(${PROJECT_NAME} PRIVATE
		"${CMAKE_SOURCE_DIR}/Code/Script/Public"
		"${CMAKE_SOURCE_DIR}/Dependencies/Include")
	target_link_libraries(${PROJECT_NAME} PRIVATE
		Core Animation RendererVK UI Entity File Input Script Physics Network Audio Spatial Threading Procedural Particle Force)

	apply_compile_settings(${PROJECT_NAME})
	set_target_properties(${PROJECT_NAME} PROPERTIES FOLDER Engine)
else()
	# Static scripts off: nothing to compile. An INTERFACE library links cleanly with no source, so App can name
	# AppScripts unconditionally without a stub .cpp.
	add_library(${PROJECT_NAME} INTERFACE)
endif()
